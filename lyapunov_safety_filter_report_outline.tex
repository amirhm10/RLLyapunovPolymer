\begin{abstract}
This report documents a general Lyapunov safety filter architecture that operates around a moving equilibrium computed from an offset-free target selector. The framework is controller-agnostic: a candidate action may be proposed by reinforcement learning, model predictive control, a residual policy, or another control law. A Lyapunov decrease test is applied to the candidate action and, when the condition is not satisfied, an optimization-based correction step computes a minimally invasive safe alternative. This document summarizes the modeling assumptions, mathematical formulation, implementation structure, and recommended diagnostics.
\end{abstract}

\section{Disturbance-Augmented Model and Observer}
\begin{align}
z_{k+1} &= A_{\mathrm{aug}} z_k + B_{\mathrm{aug}} u_k, \\
y_k &= C_{\mathrm{aug}} z_k,
\end{align}
with
\begin{align}
z_k = \begin{bmatrix} x_k \\ d_k \end{bmatrix},
\end{align}
and block structure
\begin{align}
A_{\mathrm{aug}} = \begin{bmatrix} A & B_d \\ 0 & I \end{bmatrix}, \quad
B_{\mathrm{aug}} = \begin{bmatrix} B \\ 0 \end{bmatrix}, \quad
C_{\mathrm{aug}} = \begin{bmatrix} C & D_d \end{bmatrix}.
\end{align}

\section{Steady-State Target Selector}
\begin{align}
(I - A)x_s - Bu_s - B_d \hat d_k &= 0, \\
y_s &= Cx_s + D_d \hat d_k.
\end{align}
Discuss Stage A exact target solving and Stage B closest-admissible fallback.

\section{Equilibrium-Centered Lyapunov Function}
Define
\begin{align}
e_x(k) &= \hat x_k - x_s(k), \\
V_k &= e_x(k)^\top P_x e_x(k).
\end{align}

\section{Candidate-Action Acceptance Rule}
For a candidate action $u_{\mathrm{cand}}$,
\begin{align}
e_{x, k+1|k}(u_{\mathrm{cand}}) &= A e_x(k) + B\big(u_{\mathrm{cand}} - u_s(k)\big), \\
V_{k+1|k}(u_{\mathrm{cand}}) &= e_{x, k+1|k}(u_{\mathrm{cand}})^\top P_x e_{x, k+1|k}(u_{\mathrm{cand}}).
\end{align}
Accept if
\begin{align}
V_{k+1|k}(u_{\mathrm{cand}}) \le \rho V_k + \varepsilon_{\mathrm{lyap}}.
\end{align}

\section{Optimization-Based Correction Step}
\begin{align}
\min_{u, s_v, s_u} \quad & (u - u_{\mathrm{cand}})^\top W_u (u - u_{\mathrm{cand}}) + \lambda_v s_v^2 + \lambda_u s_u^2 \\
\text{s.t.} \quad & u_{\min} \le u \le u_{\max}, \\
& \Delta u_{\min} \le u - u_{k-1} \le \Delta u_{\max}, \\
& V_{k+1|k}(u) \le \rho V_k + \varepsilon_{\mathrm{lyap}} + s_v, \\
& |u - u_{\mathrm{cand}}| \le \delta_{\mathrm{trust}} + s_u, \\
& s_v \ge 0, \quad s_u \ge 0.
\end{align}
